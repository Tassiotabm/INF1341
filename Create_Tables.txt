 CREATE TABLE PARTICIPANTE
       (        ID_PARTICIPANTE NUMBER NOT NULL,
        NOME VARCHAR(25) NOT NULL,
        NACIONALIDADE VARCHAR(25) NOT NULL,
        DATANASC DATE
        );
ALTER TABLE PARTICIPANTE ADD CONSTRAINT XPKPARTICIPANTE PRIMARY KEY (ID_PARTICIPANTE);
  
  CREATE TABLE MODALIDADE
       (        ID_MODALIDADE NUMBER NOT NULL,
        NOME VARCHAR(25) NOT NULL,
        DISTANCIA NUMBER,
        SEXO VARCHAR(25) NOT NULL
        );
ALTER TABLE MODALIDADE ADD CONSTRAINT XPKMODALIDADE PRIMARY KEY (ID_MODALIDADE);

  CREATE TABLE TORNEIO
       (        ID_TORNEIO NUMBER NOT NULL,
        ID_MODALIDADE NUMBER NOT NULL,
        NOME VARCHAR(25) NOT NULL,
        DIFICULDADE NUMBER
        );
ALTER TABLE TORNEIO ADD CONSTRAINT XPKTORNEIO PRIMARY KEY (ID_TORNEIO,ID_MODALIDADE);
ALTER TABLE TORNEIO ADD CONSTRAINT XFKTORNEIO FOREIGN KEY (ID_MODALIDADE) REFERENCES MODALIDADE (ID_MODALIDADE);

  CREATE TABLE SERIE
       (        ID_SERIE NUMBER NOT NULL,
        ID_MODALIDADE NUMBER NOT NULL,
        STATUS VARCHAR(25) NOT NULL,
        DATAINI DATE
        );
        
ALTER TABLE SERIE ADD CONSTRAINT XPKSERIE PRIMARY KEY (ID_SERIE,ID_MODALIDADE);
ALTER TABLE SERIE ADD CONSTRAINT XFKSERIE FOREIGN KEY (ID_MODALIDADE) REFERENCES MODALIDADE (ID_MODALIDADE);


  CREATE TABLE INSCRITO
       (       ID_PARTICIPANTE NUMBER NOT NULL,
        ID_MODALIDADE NUMBER NOT NULL,
        ID_TORNEIO NUMBER NOT NULL,
        MARCA NUMBER
        );
        
ALTER TABLE INSCRITO ADD CONSTRAINT XPKINSCRITO PRIMARY KEY (ID_PARTICIPANTE,ID_MODALIDADE,ID_TORNEIO);
ALTER TABLE INSCRITO ADD CONSTRAINT XFKINSCRITOPART FOREIGN KEY (ID_PARTICIPANTE) REFERENCES PARTICIPANTE (ID_PARTICIPANTE);
ALTER TABLE INSCRITO ADD CONSTRAINT XFKINSCRITOTORNEIO FOREIGN KEY (ID_TORNEIO,ID_MODALIDADE) REFERENCES TORNEIO (ID_TORNEIO,ID_MODALIDADE);

  CREATE TABLE ALOCADO
       (       ID_PARTICIPANTE NUMBER NOT NULL,
        ID_SERIE NUMBER NOT NULL,
        ID_MODALIDADE NUMBER NOT NULL,
        NUMERO_PART NUMBER NOT NULL,
        RESULTADO NUMBER
        );
ALTER TABLE ALOCADO ADD CONSTRAINT XPKALOCADO PRIMARY KEY (ID_PARTICIPANTE,ID_SERIE,ID_MODALIDADE);
ALTER TABLE ALOCADO ADD CONSTRAINT XFKALOCADOPART FOREIGN KEY (ID_PARTICIPANTE) REFERENCES PARTICIPANTE (ID_PARTICIPANTE);
ALTER TABLE ALOCADO ADD CONSTRAINT XFKALOCADOSERIE FOREIGN KEY (ID_SERIE,ID_MODALIDADE) REFERENCES SERIE (ID_SERIE,ID_MODALIDADE);
